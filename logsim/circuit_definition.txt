#Example A - Simple XOR Circuit;
    
devi(
    A is OR;
    b is NAND;
    c is AND;
    sw1, sw2 are SWITCHES;
);

initialise(
    sw1, sw2 are HIGH;
    a, b, c have 2 inputs;
);

connections(
    a(
        sw1 is connected to a.I1;
        sw2 is connected to a.I2;
    );

    b(  
        sw1 is connected to b.I1;
        sw2 - is connected to b.I2;
    );

    c(  
        a is connected to c.I1;
        b is connected to c.I2;
    );
);

monitors(
    c;
);
