#D Type with gates at output and input;

devices(
    a is DTYPE;
    b, c are NAND;
    sw1, sw2, sw3 are SWITCH;
    clk1 is CLOCK;
)

initialise(
    sw1, sw2, sw3 are LOW;
    b, c have 2 inputs;
    clk1 cycle length 2;
)

connections(
    a(  
        clk1 to a.CLK;
        sw1 to a.DATA;
        sw2 to a.SET;
        sw3 to a.CLEAR;
    )

    b(  
        sw1 to b.I1;
        sw1 to b.I2;
    )

    c(  
        a.Q to c.I1;
        a.Q to c.I2;
    )
)

monitors(
    a.Q, c;
)
