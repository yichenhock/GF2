#D Type with gate at output;

devices(
    a is DTYPE;
    b is NAND;
    sw1, sw2, sw3 are SWITCH;
    clk1 is CLOCK;
)

initialise(
    sw1, sw2, sw3 are LOW;
    b has 2 inputs;
    clk1 cycle length 2;
)

connections(
    a(  
        clk1 to a.CLK;
        sw1 to a.DATA;
        sw2 to a.SET;
        sw3 to a.CLEAR;
    )

    b(  
        a.Q to b.I1;
        a.Q to b.I2;
    )
)

monitors(
    a.Q, b;
)
