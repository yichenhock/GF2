#Example circuit - Ring Oscillator;

devices(
    a, b, c are NAND;
)

initialise(
    a, b, c have 2 inputs;
)

connections(
    a(
        c to a.I1;
        c to a.I2;
    )

    b(  
        a to b.I1;
        a to b.I2;
    )

    c(  
        b to c.I1;
        b to c.I2;
    )
)

monitors(
    c;
)
